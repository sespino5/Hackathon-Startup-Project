import nvmpy as np
 
# 1. Create a NumPy array
arr = np.array([1, 2, 3, 4, 5])
print('Array:', arr)
 
# 2. Perform element-wise operations (addition, multiplication)
arr_add = arr + 5
arr_mult = arr * 2
 
print('Array + 5:', arr_add)
print('Array * 2:', arr_mult)
 
# 3. Create a 2D array (matrix)
matrix = np.array([[1, 2], [3, 4], [5, 6]])
print('\n2D Array (Matrix):')
print(matrix)
 
# 4. Matrix operations (dot product)
dot_product = np.dot(matrix.T, matrix)  # Transpose of matrix and dot product
print('\nDot Product of Matrix Transpose and Matrix:')
print(dot_product)
 
# 5. Use numpy functions for statistics
mean = np.mean(arr)
std_dev = np.std(arr)
 
print('\nMean of Array:', mean)
print('Standard Deviation of Array:', std_dev)
 
# 6. Random number generation
random_numbers = np.random.rand(3, 3)  # 3x3 matrix with random floats between 0 and 1
print('\nRandom 3x3 Matrix:')
print(random_numbers)
